int login()
{
    char user[20], pass[20];
    int count = 3;

    while(count > 0)
    {
        printf("####################################\n");
        printf("        LIBRARY LOGIN SYSTEM         \n");
        printf("####################################\n");
        
        printf("Username: ");
        scanf("%s", user);

        printf("Password: ");
        scanf("%s", pass);


        if(strcmp(user, "admin") == 0 && strcmp(pass, "1234") == 0)
        {
            printf("Login Successful! Welcome Admin\n");
            return 1;   
        }

        else if(strcmp(user, "student") == 0 && strcmp(pass, "1111") == 0)
        {
            printf("Login Successful! Welcome Student\n");
            return 2;   
        }

        else
        {
            count--;
            printf("Invalid Login! Attempts left: %d\n", count);
        }
    }

    printf("Too many failed attempts! Exiting...\n");
    return 0;
}
void insert()
{

    printf("Enter Book ID        : ");

int count()
{
    FILE *fp;
    struct library s;
    int c=0;

    fp=fopen("books.dat","rb");
    if(fp==NULL) return 0;

    while(fread(&s,sizeof(s),1,fp)) c++;

    fclose(fp);
    return c;
}

void sort()
{
    FILE *fp;
    struct library *p, temp;
    int total = count();

    if (total == 0)
    {
        printline();
        printf(" No records found!\n");
        printline();       
        return;
    }

    p = (struct library*)malloc(total * sizeof(struct library));
    if (p == NULL)
    {
        printf(" Memory allocation failed!\n");
        exit(1);
    }

    fp = fopen("books.dat", "rb");
    if (fp == NULL)
    {
        printf(" Error opening file!\n");
        exit(1);
    }

    for (int i = 0; i < total; i++)
        fread(&p[i], sizeof(struct library), 1, fp);

    fclose(fp);

    int choice;

    do
    {
        printline();
        printf("            SORT MENU \n");
        printline();
        printf("1. Sort by ID\n");
        printf("2. Sort by Title\n");
        printf("3. Sort by Author\n");
        printf("4. Sort by Price\n");
        printf("5. Sort by Year\n");
        printf("6. Sort by Pages\n");
        printf("7. Sort by Quantity\n");
        printf("8. Sort by Category\n");   
        printf("9. Exit\n");               
        printline();

        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice)
        {
            case 1: 
                for (int i = 0; i < total - 1; i++)
                    for (int j = 0; j < total - i - 1; j++)
                        if (p[j].id > p[j+1].id)
                        {
                            temp = p[j];
                            p[j] = p[j+1];
                            p[j+1] = temp;
                        }
                break;

            case 2: 
                for (int i = 0; i < total - 1; i++)
                    for (int j = 0; j < total - i - 1; j++)
                        if (strcmp(p[j].title, p[j+1].title) > 0)
                        {
                            temp = p[j];
                            p[j] = p[j+1];
                            p[j+1] = temp;
                        }
                break;

            case 3: 
                for (int i = 0; i < total - 1; i++)
                    for (int j = 0; j < total - i - 1; j++)
                        if (strcmp(p[j].book_author, p[j+1].book_author) > 0)
                        {
                            temp = p[j];
                            p[j] = p[j+1];
                            p[j+1] = temp;
                        }
                break;

            case 4: 
                for (int i = 0; i < total - 1; i++)
                    for (int j = 0; j < total - i - 1; j++)
                        if (p[j].price > p[j+1].price)
                        {
                            temp = p[j];
                            p[j] = p[j+1];
                            p[j+1] = temp;
                        }
                break;

            case 5: 
                for (int i = 0; i < total - 1; i++)
                    for (int j = 0; j < total - i - 1; j++)
                        if (p[j].year > p[j+1].year)
                        {
                            temp = p[j];
                            p[j] = p[j+1];
                            p[j+1] = temp;
                        }
                break;

            case 6: 
                for (int i = 0; i < total - 1; i++)
                    for (int j = 0; j < total - i - 1; j++)
                        if (p[j].pages > p[j+1].pages)
                        {
                            temp = p[j];
                            p[j] = p[j+1];
                            p[j+1] = temp;
                        }
                break;

            case 7: 
                for (int i = 0; i < total - 1; i++)
                    for (int j = 0; j < total - i - 1; j++)
                        if (p[j].quantity > p[j+1].quantity)
                        {
                            temp = p[j];
                            p[j] = p[j+1];
                            p[j+1] = temp;
                        }
                break;

            case 8: 
                for (int i = 0; i < total - 1; i++)
                    for (int j = 0; j < total - i - 1; j++)
                        if (strcmp(p[j].category, p[j+1].category) > 0)
                        {
                            temp = p[j];
                            p[j] = p[j+1];
                            p[j+1] = temp;
                        }
                break;

            case 9:
                printf("\n Exiting Sort Menu...\n");
                break;

            default:
                printf("\n Invalid choice!\n");
        }

        if (choice >= 1 && choice <= 8)
        {
            printline();
            printf("                               SORTED RECORDS \n");
            printline();
            printf("%-5s %-25s %-25s %-10s %-8s %-15s %-8s %-8s\n",
                   "ID","TITLE","AUTHOR","PRICE","PGS","CATEGORY","YEAR","QTY");

            printline();

            for (int i = 0; i < total; i++)
            {
                printf("%-5d %-25.25s %-25.25s %-10.2f %-8d %-15.15s %-8d %-8d\n",
                       p[i].id,
                       p[i].title,
                       p[i].book_author,
                       p[i].price,
                       p[i].pages,
                       p[i].category,
                       p[i].year,
                       p[i].quantity);
            }
            printline();
        }

    } while (choice != 9);

    free(p);
}

            printf("\n Record not found!\n");

void sort_member()
{
    FILE *fp;
    struct member *p, temp;
    int total = 0;

    
    fp = fopen("members.dat", "rb");
    if(fp == NULL)
    {
        printf("No records found!\n");
        return;
    }

    while(fread(&temp, sizeof(temp), 1, fp))
        total++;

    fclose(fp);

    if(total == 0)
    {
        printf("No records found!\n");
        return;
    }

    
    p = (struct member*)malloc(total * sizeof(struct member));
    if(p == NULL)
    {
        printf("Memory allocation failed\n");
        exit(1);
    }

   
    fp = fopen("members.dat", "rb");
    if(fp == NULL)
    {
        printf("Error opening file\n");
        exit(1);
    }

    for(int i = 0; i < total; i++)
        fread(&p[i], sizeof(struct member), 1, fp);

    fclose(fp);

    int choice;

    do
    {
        printf("\n------ MEMBER SORT MENU ------\n");
        printf("1. Sort by Member ID\n");
        printf("2. Sort by Name\n");
        printf("3. Sort by Address\n");
        printf("4. Sort by Contact\n");
        printf("5. Exit\n");
        printf("Enter choice: ");
        scanf("%d", &choice);

        switch(choice)
        {
            case 1: 
                for(int i = 0; i < total - 1; i++)
                    for(int j = 0; j < total - i - 1; j++)
                        if(p[j].member_id > p[j+1].member_id)
                        {
                            temp = p[j];
                            p[j] = p[j+1];
                            p[j+1] = temp;
                        }
                break;

            case 2: 
                for(int i = 0; i < total - 1; i++)
                    for(int j = 0; j < total - i - 1; j++)
                        if(strcmp(p[j].name, p[j+1].name) > 0)
                        {
                            temp = p[j];
                            p[j] = p[j+1];
                            p[j+1] = temp;
                        }
                break;

            case 3: 
                for(int i = 0; i < total - 1; i++)
                    for(int j = 0; j < total - i - 1; j++)
                        if(strcmp(p[j].address, p[j+1].address) > 0)
                        {
                            temp = p[j];
                            p[j] = p[j+1];
                            p[j+1] = temp;
                        }
                break;

            case 4: 
                for(int i = 0; i < total - 1; i++)
                    for(int j = 0; j < total - i - 1; j++)
                        if(strcmp(p[j].contact, p[j+1].contact) > 0)
                        {
                            temp = p[j];
                            p[j] = p[j+1];
                            p[j+1] = temp;
                        }
                break;

            case 5:
                printf("Exiting Sort Menu...\n");
                break;

            default:
                printf("Invalid choice!\n");
        }

        if(choice >= 1 && choice <= 4)
        {
            printf("\n------ SORTED MEMBERS ------\n");

            printf("%-10s %-25s %-35s %-15s\n",
       "ID","NAME","ADDRESS","CONTACT");

            printline();

            for(int i = 0; i < total; i++)
{
    printf("%-10d %-25.25s %-35.35s %-15s\n",
           p[i].member_id,
           p[i].name,
           p[i].address,
           p[i].contact);
}

printline();
        }

    } while(choice != 5);

    free(p);
}

void issue_book()
{
    FILE *fb,*fi,*temp;
    struct library b;
    struct issue i;

    int bid, mid;
    int found = 0;

    fb = fopen("books.dat","r");
    temp = fopen("temp.dat","w");

    if(fb==NULL || temp==NULL)
    {
        printf("opening file error\n");
        return;
    }

    printf("Enter Book ID to issue = ");
    scanf("%d",&bid);

    printf("Enter Member ID = ");
    scanf("%d",&mid);

    while(fread(&b,sizeof(b),1,fb))
    {
        if(b.id == bid)
        {
            found = 1;

            if(b.quantity <= 0)
            {
                printf("\nNo copies available for this book.\n");
                fclose(fb);
                fclose(temp);
                return;
            }

            b.quantity--;   // reduce quantity
        }
        fwrite(&b,sizeof(b),1,temp);
    }

    fclose(fb);
    fclose(temp);

    fb = fopen("books.dat","w");
    temp = fopen("temp.dat","r");

    while(fread(&b,sizeof(b),1,temp))
    {
        fwrite(&b,sizeof(b),1,fb);
    }

    fclose(fb);
    fclose(temp);

    if(!found)
    {
        printf("\nBook not found.\n");
        return;
    }

    fi = fopen("issues.dat","a");

    if(fi == NULL)
    {
        printf("Issue file error\n");
        return;
    }

    i.book_id = bid;
    i.member_id = mid;

    printf("Enter Member Name: ");
    scanf(" %[^\n]", i.member_name);

    printf("Enter Issue Date: ");
    scanf("%s", i.issue_date);

    strcpy(i.return_date, "NotReturned");
    i.returned = 0;

    fwrite(&i,sizeof(i),1,fi);

    fclose(fi);

    printf("\nBook issued successfully.\n");
}
void return_book()
{

int isValidMobile(char mob[])
{
    int i = 0;

    // length must be 10
    while(mob[i] != '\0') i++;
    if(i != 10) return 0;

    // check all digits
    for(i = 0; mob[i] != '\0'; i++)
    {
        if(mob[i] < '0' || mob[i] > '9')
            return 0;
    }
    return 1;
}
